public class Example6 {

	String[] resolvers;
	int size;

	/* Found in Tomcat
	https://github.com/apache/tomcat/pull/13/commits
	*/
    public void add(String elResolver) {
        if (resolvers.length < size) {
            resolvers[size] = elResolver;
        } else {
            String[] nr = new String[size + 1];
            System.arraycopy(resolvers, 0, nr, 0, size);
            nr[size] = elResolver;
            resolvers = nr;
        }
        size ++;
    }
}
